double s(int n, int a[], int x) {
double s = a[0];
for (int i = 1; i < n; i++) {
s += a[i] * pow(x, i);
}
return s;
}
double srek(int n, int a[], int x) {
if (n == 0) return a[0]-1;
else return srek(n - 1, a, x) + a[n-1] * pow(x, n-1);
}
